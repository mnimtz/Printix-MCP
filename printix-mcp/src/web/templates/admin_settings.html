App erstellen per Azure CLI
Führe das Script in PowerShell oder Bash aus — die Zugangsdaten werden als JSON ausgegeben. Voraussetzung: Azure CLI 
# Printix MCP — Entra App erstellen (PowerShell)
az login
$app = az ad app create `
  --display-name "Printix MCP Server" `
  --sign-in-audience AzureADMultipleOrgs `
  --web-redirect-uris "{{ entra_redirect_uri }}" | ConvertFrom-Json
$secret = az ad app credential reset `
  --id $app.appId --display-name "Printix MCP" `
  --years 2 | ConvertFrom-Json
@{tenant_id=(az account show --query tenantId -o tsv)
  client_id=$app.appId
  client_secret=$secret.password} | ConvertTo-Json
# Printix MCP — Entra App erstellen (Bash)
az login
APP=$(az ad app create \
  --display-name "Printix MCP Server" \
  --sign-in-audience AzureADMultipleOrgs \
  --web-redirect-uris "{{ entra_redirect_uri }}" -o json)
APP_ID=$(echo "$APP" | jq -r '.appId')
SECRET=$(az ad app credential reset \
  --id "$APP_ID" --display-name "Printix MCP" \
  --years 2 -o json)
echo "{\"tenant_id\":\"$(az account show --query tenantId -o tsv)\",\"client_id\":\"$APP_ID\",\"client_secret\":\"$(echo "$SECRET" | jq -r '.password')\"}"
